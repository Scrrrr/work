シェルプロンプトテスト

テスト1: 基本的なシェルプロンプト

root@u3-tr201:~# mkdir -p /etc/systemd/timesyncd.conf.d/
root@u3-tr201:~# echo "test"
user@hostname:~$ ls -la

テスト2: 複数行のコマンド

root@u3-tr201:~# cat > /etc/systemd/timesyncd.conf.d/override.conf << EOF
[Time]
NTP=10.10.3.254
FallbackNTP=time.google.com
EOF
root@u3-tr201:~# systemctl restart systemd-timesyncd

テスト3: コメントを含むコマンド

root@u3-tr201:~# echo "test" # これはコメントです
root@u3-tr201:~# # これはコメント行です
root@u3-tr201:~# echo "another test"
